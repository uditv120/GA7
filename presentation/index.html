import pandas as pd

# Probability of Default (toy example)
def pd_logit(x):
    beta = np.array([ -2.1, 0.8, 1.2 ])
    z = beta @ x
    return 1 / (1 + np.exp(-z))

x = np.array([1.0, 0.3, 0.1])
